55 /***************************************************************************//**
56  * Processes each word in the buffer and directly converts it to a symbol.
57  ******************************************************************************/
58 struct Symbol* translate(char* data)
59 {
60  struct Symbol* symbols = newSymbol(BLANK);
61  // repeat until \0
62  while (data[0] != '\0')
63  {
64  // check for string literal
65  if (data[0] == '"')
66  {
67  // find next ", ignoring the first one
68  int length = strchr(data + 1, '"') - data - 1;
69  // create new string symbol
70  struct Symbol* symbol = newSymbol(STRING);
71  // set string literal (null terminated)
72  symbol->string = malloc(length + 1);
73  memcpy(symbol->string, data + 1, length);
74  symbol->string[length] = '\0';
75  // and the id
76  symbol->id = numStrings;
77  numStrings++;
78  // add it to the list
79  addSymbol(symbols, symbol);
80  // continue on to the next word
81  data += length + 2;
82  continue;
83  }
84  // check for character literal
85  if (data[0] == '\'')
86  { // idk what to do here...
87 // addSymbol(symbols, newSymbol(BLANK));
88  }
89  // one time redundancy, but whatever
90  enum CharType type = getType(data[0]);
91  char word[32];
92  size_t indice = 0;
93  // read characters of string
94  while (getType(data[indice]) == type)
95  {
96  word[indice] = data[indice];
97  indice++;
98  }
99  word[indice] = '\0'; // null terminate
100  // remove word from data
101  data += strlen(word);
102 
103  // process word
104  switch (type)
105  {
106  case WHITESPACE:
107  // do nothing
108  break;
109  case SPECIAL:
110  {
111  size_t id = 0; // index of operators[]
112  size_t i = 0; // indice of word
113  // sizeof arr / sizeof element -> number of elements
114  while (id < sizeof(operators) / sizeof(operators[0]) && strlen(word + i) > 0)
115  {
116  // if it is this operator
117  if (strncmp(word + i, operatorString(operators[id]), strlen(operatorString(operators[id]))) == 0)
118  {
119  // add the symbol
120  addSymbol(symbols, newSymbol(operators[id]));
121  // increase indice
122  i += strlen(operatorString(operators[id]));
123  // restart search
124  id = 0;
125  }
126  // otherwise move on to the next operator
127  else
128  id++;
129  }
130  break;
131  }
132  case NORMAL:
133  {
134  // same as with operators, but each word can only be one keyword
135  size_t keywordID = 0;
136  char found = 0;
137  while (keywordID < sizeof(keywords) / sizeof(keywords[0]))
138  {
139  // if it is this keyword
140  if (strlen(word) == strlen(keywordString(keywords[keywordID])) && strncmp(word, keywordString(keywords[keywordID]), strlen(word)) == 0)
141  {
142  // add the symbol
143  addSymbol(symbols, newSymbol(keywords[keywordID]));
144  found = 1;
145  break;
146  }
147  // otherwise, try next keyword
148  keywordID++;
149  }
150  // if it was a keyword, skip the rest
151  if (found)
152  break;
153 
154  // check number
155  int isNumber = 1;
156  int digitIndex = 0;
157  while (digitIndex < strlen(word))
158  {
159  // if not a digit, then it's not a number
160  if (word[digitIndex] < '0' || word[digitIndex] > '9')
161  {
162  isNumber = 0;
163  break;
164  }
165  digitIndex++;
166  }
167  if (isNumber && strlen(word) > 0)
168  {
169  struct Symbol* number = newSymbol(VALUE);
170  // convert to number
171  number->value = strtol(word, NULL, 10);
172  addSymbol(symbols, number);
173  break;
174  }
175 
176  // create symbol with name
177  struct Symbol* name = newSymbol(VARIABLE);
178  name->name = malloc(strlen(word) + 1);
179  strcpy(name->name, word);
180  // check if name exists
181  size_t id = 0;
182  while (id < numNames)
183  {
184  // if names are same
185  if (strcmp(word, names[id]) == 0)
186  {
187  // set id to same value
188  name->id = id;
189  break;
190  }
191  id++;
192  }
193  // if no names matched
194  if (id == numNames)
195  {
196  // create new name
197  names[id] = malloc(strlen(word) + 1);
198  strcpy(names[id], word);
199  name->id = numNames;
200  numNames++;
201  }
202  addSymbol(symbols, name);
203  break;
204  }
205  }
206  }
207  // remove blanks from beginning
208  while (symbols && symbols->type == BLANK)
209  {
210  struct Symbol* blank = symbols;
211  symbols = symbols->next;
212  deleteSymbol(blank);
213  }
214  // remove blanks after beginning
215  struct Symbol* current = symbols;
216  while (current)
217  {
218  if (current->next && current->next->type == BLANK)
219  {
220  struct Symbol* blank = current->next;
221  current->next = current->next->next;
222  deleteSymbol(blank);
223  }
224  else current = current->next;
225  }
226  return symbols;
227 }

256 /***************************************************************************//**
257  * Some operators can be used in different ways. In this stage, the context on
258  * both sides of the operator is used to determine which symbol it is.
259  ******************************************************************************/
260 struct Symbol* interpret(struct Symbol* symbols)
261 {
262  struct Symbol* current = symbols;
263  // if at beginning, not operand before
264  switch (current->type)
265  {
266  case PLUS: current->type = UPLUS; break;
267  case MINUS: current->type = UMINUS; break;
268  case ASTERISK: current->type = DEREFERENCE; break;
269  case AMPERSAND: current->type = ADDRESS; break;
270  case INCREMENT: current->type = INCREMENT_PRE; break;
271  case DECREMENT: current->type = DECREMENT_PRE; break;
272  default: break;
273  }
274  while (current && current->next)
275  {
276  switch (current->next->type)
277  {
278  // if there is an operand before +/-,
279  // then it is add/sub; otherwise it
280  // is unary plus/minus
281  case PLUS:
282  if (isOperand(current->type))
283  current->next->type = ADD;
284  else
285  current->next->type = UPLUS;
286  break;
287  case MINUS:
288  if (isOperand(current->type))
289  current->next->type = SUBTRACT;
290  else
291  current->next->type = UMINUS;
292  break;
293  // multiply has operand before
294  // dereference does not
295  case ASTERISK:
296  if (isOperand(current->type))
297  current->next->type = MULTIPLY;
298  else
299  current->next->type = DEREFERENCE;
300  break;
301  // bitwise and has operand before
302  // address does not
303  case AMPERSAND:
304  if (isOperand(current->type))
305  current->next->type = BAND;
306  else
307  current->next->type = ADDRESS;
308  break;
309  // inc/dec post has operand before
310  // inc/dec pre does not
311  case INCREMENT:
312  if (isOperand(current->type))
313  current->next->type = INCREMENT_POST;
314  else
315  current->next->type = INCREMENT_PRE;
316  break;
317  case DECREMENT:
318  if (isOperand(current->type))
319  current->next->type = DECREMENT_POST;
320  else
321  current->next->type = DECREMENT_PRE;
322  break;
323  default:
324  break;
325  }
326  current = current->next;
327  }
328  return symbols;
329 }

572 /***************************************************************************//**
573  * Searches the symbol list to find the given symbol in the given pattern. The
574  * list is traversed in the specified direction.
575  ******************************************************************************/
576 struct Symbol* findPattern(struct Symbol* symbols, enum Pattern pattern, enum Direction direction, enum SymbolType type)
577 {
578  switch (pattern)
579  {
580  case CONTAINER:
581  {
582  enum SymbolType open, close;
583  switch (type)
584  {
585  case PARENTHESES:
586  open = BPAREN;
587  close = EPAREN;
588  break;
589  case BRACKET:
590  open = BBRACK;
591  close = EBRACK;
592  break;
593  case SUBSCRIPT:
594  open = BBRACE;
595  close = EBRACE;
596  break;
597  default:
598  open = type;
599  close = type;
600  break;
601  }
602  return patternContainer(symbols, type, open, close);
603  break;
604  }
605  case UNARY_POST: return patternUnaryPost(symbols, direction, type);
606  case UNARY_PRE: return patternUnaryPre(symbols, direction, type);
607  case BINARY: return patternBinary(symbols, direction, type);
608  case TERNARY:
609  {
610  enum SymbolType first, second;
611  switch (type)
612  {
613  case TERNARY_CONDITIONAL:
614  first = QUESTION;
615  second = COLON;
616  break;
617  default:
618  first = type;
619  second = type;
620  break;
621  }
622  return patternTernary(symbols, direction, type, first, second);
623  }
624  }

627 /***************************************************************************//**
628  * Searches the symbol list for a container. These are a set of symbols that are
629  * enclosed by an opening symbol and a closing symbol.
630  ******************************************************************************/
631 struct Symbol* patternContainer(struct Symbol* symbols, enum SymbolType type, enum SymbolType open, enum SymbolType close)
632 {
633  struct Symbol* current = symbols;
634  struct Symbol* openSymbols[32];
635  int depth = 0;
636  while (current)
637  {
638  // check if it is open symbol
639  if (current->type == open)
640  {
641  openSymbols[depth] = current;
642  depth++;
643  }
644  // check if it is close symbol
645  if (current->next && current->next->type == close)
646  {
647  // connect it to last open: set rhs to inside, next to after
648  struct Symbol* container = openSymbols[depth - 1];
649  container->type = type;
650  // check if empty
651  if (current == container)
652  {
653  struct Symbol* ebrack = current->next;
654  container->next = current->next->next;
655  deleteSymbol(ebrack);
656  }
657  else
658  {
659  container->rhs = container->next;
660  container->next = current->next->next;
661  current->next = NULL;
662  current = container;
663  }
664  // process contents
665  container->rhs = combine(container->rhs);
666  depth--;
667  }
668  current = current->next;
669  }
670  return symbols;
671 }

675  * operand symbols in which the operator appears after the operand.
676  ******************************************************************************/
677 struct Symbol* patternUnaryPost(struct Symbol* symbols, enum Direction direction, enum SymbolType type)
678 {

682  {

684  struct Symbol* operand = symbols;
685  operator->lhs = operand;
686  operand->next = NULL;
687  symbols = operator;
688  }

691  {

693  struct Symbol* operand = current->next;
694  if (operator->type == type)

713  current->lhs = var;
714  current->next = var->next;

716  }

194 /***************************************************************************//**
195  * @struct Symbol
196  *
197  * @brief a symbol representing a C construct
198  *
199  * This is a tree of symbols. Each symbol has a follower and up to three
200  * arguments, as well as a type and various forms of symbol data.
201  ******************************************************************************/
202 struct Symbol
203 {
204  enum SymbolType type; /**< The type of symbol. */
205 
206  int id; /**< The identification number of the symbol. */
207  int value; /**< The value of the symbol. */
208  char* string; /**< String data for the symbol. */
209  char* name; /**< The name of the symbol. */
210 
211  struct Symbol* lhs; /**< A pointer to the left argument symbol. */
212  struct Symbol* rhs; /**< A pointer to the right argument symbol. */
213  struct Symbol* third; /**< A pointer to the third argument symbol.
214  ** Only used by the ternary conditional */
215 
216  struct Symbol* next; /**< A pointer to the next symbol in the list. */
217 };

62 /***************************************************************************//**
63  * The entire tree of symbols is compiled into assembly. Each line of assembly
64  * is stored in its own string structure.
65  ******************************************************************************/
66 struct String* compile(struct Symbol* symbols)
67 {
68  char buffer[512];
69  scope = newScope();
70  data = newString(0);
71  struct String* text = NULL;
72  while (symbols)
73  {
74  if (text == NULL)
75  text = getAssembly(symbols, VAL);
76  else
77  addString(text, getAssembly(symbols, VAL));
78  symbols = symbols->next;
79  }
80  // insert data before text
81  addString(data, text);
82  return data;
83 }

88 /***************************************************************************//**
89  * Creates assembly code for the base symbol. Other symbols in the list are
90  * ignored, but all arguments required for the base symbol are processed.
91  ******************************************************************************/
92 struct String* getAssembly(struct Symbol* symbols, enum Pass pass)
93 {
94  struct String* assembly = NULL;
95  if (symbols)
96  {
97  // create empty string
98  struct String* current = newString(0);
99  // create temporary buffer
100  char buffer[32] = "";
101  // process the current symbol
102  switch (symbols->type)
103  {
104  // results stored in %eax
105  case VARIABLE:
106  sprintf(buffer, "\t# var %i %s\n", symbols->id, symbols->name);
107  addString(current, getString(buffer));
108  if (pass == VAL)
109  sprintf(buffer, "\tmov %i(%%ebp),%%eax\n", getOffset(scope, symbols->id));
110  else
111  sprintf(buffer, "\tlea %i(%%ebp),%%eax\n", getOffset(scope, symbols->id));
112  addString(current, getString(buffer));
113  break;
114  case STRING: // strings are stored as "STR#:" where # is the id
115  addString(current, getString("\t# str\n"));
116  // store string in read only data section
117  addString(data, getString("\t.section .rodata\n"));
118  sprintf(buffer, "STR%i:\n", symbols->id);
119  addString(data, getString(buffer));
120  sprintf(buffer, "\t.string \"%s\"\n", symbols->string);
121  addString(data, getString(buffer));
122  // reference string's location
123  sprintf(buffer, "\tmov $STR%i,%%eax\n", symbols->id);
124  addString(current, getString(buffer));
125  break;
126  case VALUE:
127  addString(current, getString("\t# val\n"));
128  sprintf(buffer, "\tmov $%i,%%eax\n", symbols->value);
129  addString(current, getString(buffer));
130  break;
131  case TYPE:
132  addString(current, getString("\t# type\n"));
133  break;
134  case DECLARE:
135  sprintf(buffer, "\t# dec %i %s\n", symbols->rhs->id, symbols->rhs->name);
136  addString(current, getString(buffer));
137  addString(current, getString("\tsub $4,%esp\n"));
138  if (pass == VAL)
139  addString(current, getString("\tmov (%esp),%eax\n"));
140  else
141  addString(current, getString("\tmov %esp,%eax\n"));
142  addVariable(scope, symbols->rhs->id, 4);
143  break;
144  case PARENTHESES: // rhs: symbol
145  addString(current, getString("\t# paren\n"));
146  if (symbols->rhs)
147  addString(current, getAssembly(symbols->rhs, pass));
148  break;
149  case BRACKET: // rhs: instructions (arrays are given a separate symbol)
150  {
151  addString(current, getString("\t# brack\n"));
152  // TODO: implement scoping
153  struct Symbol* symbol = symbols->rhs;
154  while (symbol)
155  {
156  addString(current, getAssembly(symbol, VAL));
157  symbol = symbol->next;
158  }
159  break;
160  }
161  case FUNCTION: // lhs: parameters; rhs: instructions
162  {
163  addString(current, getString("\t# func\n"));
164  // header
165  addString(current, getString("\t.text\n"));
166  sprintf(buffer, "\t.globl %s\n", symbols->name);
167  addString(current, getString(buffer));
168  sprintf(buffer, "\t.type %s, @function\n", symbols->name);
169  addString(current, getString(buffer));
170  // label
171  sprintf(buffer, "%s:\n", symbols->name);
172  addString(current, getString(buffer));
173  // save stack pointer
174  addString(current, getString("\tpush %ebp\n"));
175  addString(current, getString("\tmov %esp,%ebp\n"));
176  // create scope
177  struct Scope* oldScope = scope;
178  scope = newScope();
179  // process parameters
180  struct Symbol* parameter = symbols->lhs;
181  unsigned int parameterIndex = 0;
182  while (parameter)
183  {
184  // declare: reference to var is stored in %eax and pushed
185  addString(current, getAssembly(parameter, REF));
186  addString(current, getString("\tmov %eax,%ecx\n"));
187  // evaluate argument: +8 because we push %ebp and instruction pointer
188  sprintf(buffer, "\tmov %i(%%ebp),%%eax\n", 4 * parameterIndex + 8);
189  addString(current, getString(buffer));
190  // assign
191  addString(current, getString("\tmov %eax,(%ecx)\n"));
192  parameterIndex++;
193  parameter = parameter->next;
194  }
195  // process instructions
196  struct Symbol* instruction = symbols->rhs;
197  while (instruction)
198  {
199  addString(current, getAssembly(instruction, VAL));
200  instruction = instruction->next;
201  }
202  // restore scope
203  deleteScope(scope);
204  scope = oldScope;
205  break;
206  }
207 
208  case IF: // lhs: condition; rhs: command(s)
209  {
210  deleteString(current);
211 // addString(current, getString("\t# if\n"));
212  current = getString("\t# if\n");
213  // compare to 0
214  addString(current, getAssembly(symbols->lhs, VAL));
215  addString(current, getString("\tcmp $0,%eax\n"));
216  // then skip commands if equal to 0 (false)
217  sprintf(buffer, "\tje IF%i\n", numIf);
218  addString(current, getString(buffer));
219  // the commands
220  addString(current, getAssembly(symbols->rhs, VAL));
221  // the label to skip to
222  sprintf(buffer, "IF%i:\n", numIf);
223  addString(current, getString(buffer));
224  numIf++;
225  break;
226  }
227  case WHILE: // lhs: condition; rhs: command(s)
228  {
229 // deleteString(current);
230  addString(current, getString("\t# while\n"));
231 // current = getString("\t# while\n");
232  // initial label
233  sprintf(buffer, "WHILE_BEG%i:\n", numWhile);
234  addString(current, getString(buffer));
235  // compare to 0
236  addString(current, getAssembly(symbols->lhs, VAL));
237  addString(current, getString("\tcmp $0,%eax\n"));
238  // then skip commands if equal to 0 (false)
239  sprintf(buffer, "\tje WHILE_END%i\n", numWhile);
240  addString(current, getString(buffer));
241  // the commands
242  addString(current, getAssembly(symbols->rhs, VAL));
243  //jump to beginning
244  sprintf(buffer, "\tjmp WHILE_BEG%i\n", numWhile);
245  addString(current, getString(buffer));
246  // the label to skip to
247  sprintf(buffer, "WHILE_END%i:\n", numWhile);
248  addString(current, getString(buffer));
249  numWhile++;
250  break;
251  }

292  case DEREFERENCE:
293  addString(current, getString("\t# deref\n"));
294  // evaluate operand and store in %eax
295  addString(current, getAssembly(symbols->rhs, VAL));
296  // dereference

298  addString(current, getString("\tmov (%eax),%eax\n"));
299  else // just return the address
300  {
301  }
302  break;
303  case ADDRESS:
304  addString(current, getString("\t# addr\n"));

309  // 4
310  case MULTIPLY:
311  addString(current, getString("\t# mult\n"));
312  // place lhs in %eax and rhs in %ecx
313  addString(current, getAssembly(symbols->rhs, VAL));
314  addString(current, getString("\tpush %eax\n"));
315  addString(current, getAssembly(symbols->lhs, VAL));
316  addString(current, getString("\tpop %ecx\n"));
317  addString(current, getString("\timul %ecx,%eax\n"));
318  break;
319  case DIVIDE: // a / b: lhs = a, rhs = b
320  addString(current, getString("\t# div\n"));
321  // place lhs in %eax and rhs in %ecx, %edx = 0
322  addString(current, getAssembly(symbols->rhs, VAL));
323  addString(current, getString("\tpush %eax\n"));
324  addString(current, getAssembly(symbols->lhs, VAL));
325  addString(current, getString("\tpop %ecx\n"));
326  addString(current, getString("\tmov $0,%edx\n"));
327  addString(current, getString("\tidiv %ecx\n"));
328  break;
329  case ADD:
330  addString(current, getString("\t# add\n"));
331  // place lhs in %eax and rhs in %ecx
332  addString(current, getAssembly(symbols->rhs, VAL));
333  addString(current, getString("\tpush %eax\n"));
334  addString(current, getAssembly(symbols->lhs, VAL));
335  addString(current, getString("\tpop %ecx\n"));
336  addString(current, getString("\tadd %ecx,%eax\n"));
337  break;
338  case SUBTRACT: // a - b: lhs = a, rhs = b
339  addString(current, getString("\t# sub\n"));
340  // place lhs in %eax and rhs in %ecx
341  addString(current, getAssembly(symbols->rhs, VAL));
342  addString(current, getString("\tpush %eax\n"));
343  addString(current, getAssembly(symbols->lhs, VAL));
344  addString(current, getString("\tpop %ecx\n"));
345  addString(current, getString("\tsub %ecx,%eax\n"));
346  break;
347  case ASSIGN:
348  addString(current, getString("\t# assi\n"));
349  // evaluate location and push it (to be popped into %ecx)
350  addString(current, getAssembly(symbols->lhs, REF));
351  addString(current, getString("\tpush %eax\n"));
352  // evaluate value and store it in %eax
353  addString(current, getAssembly(symbols->rhs, VAL));
354  addString(current, getString("\tpop %ecx\n"));
355  // store value in location
356  addString(current, getString("\tmov %eax,(%ecx)\n"));
357  // return the value (which is in %eax)
358  break;
359  case RETURN:
360  addString(current, getString("\t# ret\n"));
361  // if there is an operand (to the right), store it in %eax
362  if (symbols->rhs)
363  addString(current, getAssembly(symbols->rhs, VAL));
364  // reset stack pointer and return
365  addString(current, getString("\tmov %ebp,%esp\n"));
366  addString(current, getString("\tpop %ebp\n"));
367  addString(current, getString("\tret\n"));
368  break;
369  default:
370  break;
371  }
372  // clean up empty strings
373  while (current && sizeof(current->contents) == 0)
374  { // at beginning
375  struct String* blank = current;
376  current = current->next;
377  deleteString(blank);
378  }
379  struct String* strIndex = current;
380  while (strIndex && strIndex->next)
381  { // in middle
382  if (sizeof(strIndex->next->contents) == 0)
383  {
384  struct String* blank = strIndex->next;
385  strIndex->next = strIndex->next->next;
386  deleteString(blank);
387  }
388  strIndex = strIndex->next;
389  }
390  if (assembly == NULL)
391  assembly = current;
392  else
393  addString(assembly, current);
394  // process the next symbol in the next cycle
395  symbols = symbols->next;
396  }
397  return assembly;
398 }
